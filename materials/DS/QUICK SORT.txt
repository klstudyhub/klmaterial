int main()
{
    int a[100],n,i;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    quicksort(a,0,n-1);
    for(i=0;i<n;i++)
    {
        printf("%d  ",a[i]);
    }

    return 0;
}
void quicksort(int a[100],int left,int right)
{
    int i,j,pivot,temp;
    if(left<right)
    {
        i=left;
        pivot=left;
        j=right;
        while(i<j)
        {
            while(a[i]<=a[pivot])
            {
                i++;
            }
            while(a[j]>a[pivot])
            {
                j--;
            }
            if(i<j)
            {
                temp=a[i];
                a[i]=a[j];
                a[j]=temp;
            }
        }
        temp=a[j];
        a[j]=a[pivot];
        a[pivot]=temp;
   
    quicksort(a,left,j-1);
    quicksort(a,j+1,right);
     }
}
