#include<stdio.h>
#include<stdlib.h>
#define size 5
struct node
{
    int top;
    int a[size];
};
struct node s;

void push()
{
    int ele;
    printf("enter data");
    scanf("%d",&ele);
    
    if(s.top==size-1)
    printf("stack is full");
    else
    {
         s.top=s.top+1;
        s.a[s.top]=ele;
       
    }
}
void pop()
{
    if(s.top==-1)
    {
        printf("stack is empty");
        
    }
    else
    {
       
        printf("%d is deleted",s.a[s.top]);
         s.top=s.top-1;
    }
}
void display()
{
    int i;
    for(i=s.top;i>=0;i--)
    {
        printf("%d\n",s.a[i]);
    }
}
  int main()
{
    int opt;
    s.top=-1;
    while(1)
    {
        printf("\n1.push");
        printf("\n2.display");
        printf("\n3.pop");
        printf("\nenter your option");
        scanf("%d",&opt);
        switch(opt)
        {
            case 1: push();
            break;
            case 2: display();
            break;
            case 3: pop();
            break;
            case 4:exit(0);
        }
    }
}
