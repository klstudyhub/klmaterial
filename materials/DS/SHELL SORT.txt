#include <stdio.h>
void shell_sort(int a[100],int n);
int main()
{
    int a[100],n,i,gap,j,temp;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    shell_sort(a,n);
    for(i=0;i<n;i++)
    {
        printf("%d  ",a[i]);
    }
 return 0;
}
void shell_sort(int a[100],int n)
{
    int gap,i,j,temp;
    for(gap=n/2;gap>=1;gap=gap/2)
    {
        for(j=gap;j<n;j++)
        {
            for(i=j-gap;i>=0;i=i-gap)
            {
                if(a[i]>a[i+gap])
                {
                    temp=a[i];
                    a[i]=a[i+gap];
                    a[i+gap]=temp;
                }
            }
        }
    }
}
