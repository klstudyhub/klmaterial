void bubblesort(int a[],int n)
{
    int p,temp,i;
    for(p=1;p<n;p++)
    {
          for(i=0;i<n-p;i++)
          {
              if(a[i]>a[i+1])
              {
                  temp=a[i];
                  a[i]=a[i+1];
                  a[i+1]=temp;
              }
          }
    }
                
}
