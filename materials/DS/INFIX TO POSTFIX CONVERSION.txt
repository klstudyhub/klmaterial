#include<stdio.h>
#include<ctype.h>
char postfix[100];
char a[100];
int top=-1;
void push(char ch);
char pop();
int priority(char ch);

int main()
{
    int i=0,ele,k=0;
    char infix[100],ch;
    printf("enter infix expression");
    scanf("%s",infix);
    push('#');
    while((ch=infix[i++])!='\0')
    {
        if(ch=='(')
        {
            push(ch);
        }
       
        else if(ch==')')
        {
            while(a[top]!='(')
            {
                postfix[k++]=pop();
               
            }
            ele=pop();
        }
        else if(isalnum(ch))
        {
            postfix[k++]=ch;
        }
        else
        {
            while(priority(a[top])>=priority(ch))
            {
                postfix[k++]=pop();
               
            }
            push(ch);
        }
     }
     while(a[top]!='#')
     
         postfix[k++]=pop();
         postfix[k]='\0';
     
     printf("\n postfix expression is=%s",postfix);
   
}
void push(char ch)
{
    a[++top]=ch;
}
char pop()
{
    return(a[top--]);
}
int priority(char ch)
{
    if(ch=='^')
    {
        return(3);
    }
    else if(ch=='*'|| ch=='/')
    {
       return (2);
    }
    else if(ch=='+'|| ch=='-')
    {
        return (1);
    }
    else
    {
       return(0);
    }
   
}
